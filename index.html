        function cargarHistorial() {

            const hoy = new Date().toDateString();

            const registrosHoy = registros.filter(r => 

                new Date(r.fecha).toDateString() === hoy && r.empleado === currentUser

            ).sort((a, b) => b.timestamp - a.timestamp);

            

            const container = document.getElementById('historialList');

            

            if (registrosHoy.length === 0) {

                container.innerHTML = `

                    <div class="empty-state">

                        <div class="empty-state-icon">📋</div>

                        <p>No hay registros para hoy</p>

                    </div>

                `;

                document.getElementById('totalHoras').textContent = '0h 0m';

                return;

            }

            

            let html = '';

            let totalMinutos = 0;

            let entradaActual = null;

            

            const registrosOrdenados = [...registrosHoy].sort((a, b) => a.timestamp - b.timestamp);

            

            for (let i = 0; i < registrosOrdenados.length; i++) {

                const reg = registrosOrdenados[i];

                if (reg.tipo === 'entrada') {

                    entradaActual = reg.timestamp;

                } else if (reg.tipo === 'salida' && entradaActual) {

                    totalMinutos += (reg.timestamp - entradaActual) / (1000 * 60);

                    entradaActual = null;

                }

            }

            

            if (isWorking && lastEntry) {

                totalMinutos += (Date.now() - parseInt(lastEntry)) / (1000 * 60);

            }

            

            const horas = Math.floor(totalMinutos / 60);

            const minutos = Math.floor(totalMinutos % 60);

            document.getElementById('totalHoras').textContent = `${horas}h ${minutos}m`;

            

            registrosHoy.forEach(reg => {

                const hora = new Date(reg.fecha).toLocaleTimeString('es-ES');

                const esEntrada = reg.tipo === 'entrada';

                

                html += `

                    <div class="history-item ${esEntrada ? '' : 'out'}">

                        <div class="history-info">

                            <h4>${esEntrada ? '🟢 Entrada' : '🔴 Salida'}</h4>

                            ${reg.notas ? `<p>📝 ${reg.notas}</p>` : ''}

                        </div>

                        <div class="history-time">${hora}</div>

                    </div>

                `;

            });

            

            container.innerHTML = html;

        }

        function filtrarDatos() {

            const empleado = document.getElementById('filterEmpleado').value;

            const desde = document.getElementById('filterDesde').value;

            const hasta = document.getElementById('filterHasta').value;

            

            let filtrados = [...registros];

            

            if (empleado) filtrados = filtrados.filter(r => r.empleado === empleado);

            if (desde) {

                const d = new Date(desde);

                filtrados = filtrados.filter(r => new Date(r.fecha) >= d);

            }

            if (hasta) {

                const h = new Date(hasta);

                h.setHours(23, 59, 59);

                filtrados = filtrados.filter(r => new Date(r.fecha) <= h);

            }

            

            filtrados.sort((a, b) => b.timestamp - a.timestamp);

            renderizarTabla(filtrados);

            actualizarStats(filtrados);

        }


        function renderizarTabla(datos) {

            const tbody = document.getElementById('adminTableBody');

            

            if (datos.length === 0) {

                tbody.innerHTML = '<tr><td colspan="6" style="text-align: center; padding: 40px;">No hay registros para los filtros seleccionados</td></tr>';

                return;

            }

            

            tbody.innerHTML = datos.map(reg => {

                const fecha = new Date(reg.fecha);

                const esEntrada = reg.tipo === 'entrada';

                

                let duracion = '-';

                if (!esEntrada && reg.entradaRef) {

                    const minutos = (reg.timestamp - reg.entradaRef) / (1000 * 60);

                    const h = Math.floor(minutos / 60);

                    const m = Math.floor(minutos % 60);

                    duracion = `${h}h ${m}m`;

                }

                

                return `

                    <tr>

                        <td>${fecha.toLocaleDateString('es-ES')}</td>

                        <td><strong>${reg.empleado}</strong></td>

                        <td><span class="badge ${esEntrada ? 'badge-in' : 'badge-out'}">

                            ${esEntrada ? 'ENTRADA' : 'SALIDA'}

                        </span></td>

                        <td>${fecha.toLocaleTimeString('es-ES')}</td>

                        <td class="note-text" title="${reg.notas || ''}">

                            ${reg.notas || '-'}

                        </td>

                        <td>${duracion}</td>

                    </tr>

                `;

            }).join('');

        }


        function actualizarStats(datos) {

            const empleados = new Set(datos.map(r => r.empleado));

            const hoy = new Date().toDateString();

            const registrosHoy = datos.filter(r => new Date(r.fecha).toDateString() === hoy);

            

            let totalMinutos = 0;

            datos.forEach(reg => {

                if (reg.tipo === 'salida' && reg.entradaRef) {

                    totalMinutos += (reg.timestamp - reg.entradaRef) / (1000 * 60);

                }

            });

            

            document.getElementById('totalEmpleados').textContent = empleados.size;

            document.getElementById('totalRegistros').textContent = registrosHoy.length;

            document.getElementById('horasTotales').textContent = Math.floor(totalMinutos / 60) + 'h';

        }

        function exportarExcel() {

            const datos = obtenerDatosFiltrados();

            

            let csv = 'Fecha;Empleado;Tipo;Hora;Notas;Duracion\n';

            

            datos.forEach(reg => {

                const fecha = new Date(reg.fecha);

                const hora = fecha.toLocaleTimeString('es-ES');

                const fechaStr = fecha.toLocaleDateString('es-ES');

                

                let duracion = '';

                if (reg.tipo === 'salida' && reg.entradaRef) {

                    const mins = (reg.timestamp - reg.entradaRef) / (1000 * 60);

                    duracion = `${Math.floor(mins / 60)}h ${Math.floor(mins % 60)}m`;

                }

                

                csv += `${fechaStr};${reg.empleado};${reg.tipo};${hora};"${reg.notas || ''}";${duracion}\n`;

            });

            

            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement('a');

            link.href = URL.createObjectURL(blob);

            link.download = `registro-jornada-${new Date().toISOString().split('T')[0]}.csv`;

            link.click();

        }


        function obtenerDatosFiltrados() {

            let filtrados = [...registros];

            

            if (empleado) filtrados = filtrados.filter(r => r.empleado === empleado);

            if (desde) {

                const d = new Date(desde);

                filtrados = filtrados.filter(r => new Date(r.fecha) >= d);

            }

            if (hasta) {

                const h = new Date(hasta);

                h.setHours(23, 59, 59);

                filtrados = filtrados.filter(r => new Date(r.fecha) <= h);

            }

            

            return filtrados.sort((a, b) => b.timestamp - a.timestamp);

        }
